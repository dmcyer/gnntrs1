  (a)
  (eq)
  (f)

  (p)
  (q)
  (b)
  (g)
  (k)
  (true)

  a -> b | b == b
  eq(x,x) -> true
  f(x) -> g(x) | h(f(x)) == k(g(b))
  h(f(a)) -> p(a) | b == b
  h(x) -> k(x) | b == b
  p(b) -> k(f(a))
  p(x) -> q(x) | eq(x,a) == true
  q(b) -> k(g(a))
)
