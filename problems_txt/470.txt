  P : o -> (o -> o) -> (o -> o) -> o

  R : o -> o

  Z'' : o -> o

  P (L Z) (\x. Z'(x)) (\x. Z''(x)) -> Z'(Z),
  P (R Z) (\x. Z'(x)) (\x. Z''(x)) -> Z''(Z)
