(CONDITIONTYPE ORIENTED)
(VAR x y z)
(REPLACEMENT-MAP
  (lte 1, 2)
  (0)
  (false)
  (s)
  (true)
)
(RULES
  lte(0,x) -> true
  lte(s(x),0) -> false
  lte(s(x),s(y)) -> lte(x,y)
  lte(x,s(x)) -> true
  lte(x,s(y)) -> true | lte(x,y) == true
  lte(x,x) -> true
  lte(x,z) -> true | lte(x,y) == true, lte(y,z) == true
  lte(y,x) -> true | lte(x,y) == false
)
